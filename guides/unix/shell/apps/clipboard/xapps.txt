#xclip
    //-f: also print out, -r: strip newline, -t image/png: specify type of data
    xclip -o [-selection primary | -selection p]? //primary
    xclip -o -selection [secondary|sec] //use secondary
    xclip -o -selection [clipboard|clip] //use secondary

#xsel
    xsel -p //primary
    xsel -s //secondary
    xsel -b //clipboard
